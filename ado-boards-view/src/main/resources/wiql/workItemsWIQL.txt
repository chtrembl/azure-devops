SELECT 
        [System.Id],
        [System.WorkItemType],
        [System.Title],
        [System.State],
        [System.AreaPath],
        [System.AssignedTo],
        [System.TeamProject],
        [System.CreatedDate],
        [System.ChangedDate] 
FROM 
        workitems 
WHERE 
        [System.TeamProject] = @project 
AND 
        ([System.WorkItemType] == 'Bug' OR [System.WorkItemType] == 'Epic' OR [System.WorkItemType] == 'Feature' OR [System.WorkItemType] == 'Issue' OR [System.WorkItemType] == 'Task' OR [System.WorkItemType] == 'Test Case' OR [System.WorkItemType] == 'Test Plan' OR [System.WorkItemType] == 'Test Suite' OR [System.WorkItemType] == 'User Story')
AND 
        ([System.CreatedDate] > @StartOfDay('%s') OR [System.ChangedDate] > @StartOfDay('%s')) 
ORDER BY 
        [System.ChangedDate] DESC
