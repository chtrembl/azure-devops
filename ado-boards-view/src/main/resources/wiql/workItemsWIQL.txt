SELECT 
        [System.Id],
        [System.WorkItemType],
        [System.CreatedDate],
        [System.ChangedDate] 
FROM 
        workitems 
WHERE 
        [System.TeamProject] = @project 
AND 
		%s
AND 
        ([System.CreatedDate] > @StartOfDay('%s') OR [System.ChangedDate] > @StartOfDay('%s')) 
ORDER BY 
        [System.ChangedDate] DESC
